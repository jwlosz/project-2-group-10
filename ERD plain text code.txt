Table category {
  category_id varchar(10)
  category varchar(30)
}

Table subcategory {
  subcategory_id varchar (30)
  subcategory varchar(30)
}

Table campaign {
  cf_id int
  contact_id int
  company_name varchar(50)
  description varchar(50)
  goal int
  pledged int
  outcome varchar(15)
  backers_count int
  country varchar(5)
  currency varchar(5)
  launched_date datetime
  end_date datetime
  category_id varchar(30)
  subcategory_id varchar(30)
}

Table contacts {
  contact_id int
  first_name varchar(10)
  last_name varchar(10)
  email varchar(50)
}


Ref: "contacts"."contact_id" < "campaign"."contact_id"

Ref: "subcategory"."subcategory_id" < "campaign"."subcategory_id"

Ref: "category"."category_id" < "campaign"."category_id"
